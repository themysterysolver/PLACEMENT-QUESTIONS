public class spiral {

    public static int diagonalSum(int N) {
        if (N == 1) return 1;

        long runner = 1;
        long sum = 1;
        long difference = 2;

        while (runner < N * N) {
            for (int i = 0; i < 4; i++) {
                runner += difference;
                if (runner > N * N) {
                    break;
                }
                sum += runner;
            }
            difference += 2;
        }

        return (int) sum;
    }

    public static void main(String[] args) {
        for(int i=0;i<20;i++){
            System.out.println("Diagonal sum for N=" + i + " is: " + diagonalSum(i));
        }
        
    }
}
